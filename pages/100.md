```sql
average_gpa_by_major
SELECT 
    s.major,
    SUM(CAST(ap.gpa AS FLOAT)) / COUNT(*) AS average_gpa
FROM
    brighthive_dev_redshift.academic_performance ap
GROUP BY 
    s.major
ORDER BY 
    average_gpa DESC;
```
