```sql entry_year_major_distribution
SELECT 
  major, 
  entry_year, 
  COUNT(*) AS student_count
FROM "workspace-database"."brighthive_dev_redshift"."students"
GROUP BY major, entry_year
ORDER BY major, entry_year;
```
