```sql students_with_a_grade
SELECT 
    s.first_name,
    s.last_name
FROM 
    "workspace-database"."brighthive_dev_redshift"."students" s
JOIN 
    "workspace-database"."brighthive_dev_redshift"."academic_performance" a
    ON s.student_id = a.student_id
WHERE 
    a.final_grade = 'A';
```
