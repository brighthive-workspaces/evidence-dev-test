```sql error_code_line
SELECT
    error_code,
    line AS error_line_count
FROM
    error_log
ORDER BY
    error_code
```

<BarChart
    data={error_code_line}
    x=error_code
    y=error_line_count
    labels=true
    colorPalette={["#1f78b4", "#33a02c", "#e31a1c"]}
    title="Error Line Distribution by Error Code"
    height=400
/>
