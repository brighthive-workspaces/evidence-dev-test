```sql sales_revenue_by_category
SELECT 
    p.category as product_category, 
    SUM(s.sales_revenue) as total_sales_revenue
FROM 
    sales_data AS s
JOIN 
    products AS p ON s.product_id = p.product_id
GROUP BY 
    product_category
ORDER BY 
    total_sales_revenue DESC;
```
