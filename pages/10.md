```sql student_attendance
SELECT
    student_name,
    ROUND((attended_classes::float / total_classes) * 100, 2) as attendance_rate
FROM
    student_attendance_records
WHERE
    academic_year = '${inputs.academic_year}'
ORDER BY
    attendance_rate DESC;
```
