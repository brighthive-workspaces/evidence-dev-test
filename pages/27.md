```sql students_attendance
SELECT
    student_name,
    attendance_rate
FROM
    student_attendance
ORDER BY
    attendance_rate DESC;
```

```jsx
<BarChart
    data={students_attendance}
    x=student_name
    y=attendance_rate
    labels=true
    colorPalette={["#4e79a7", "#f28e2b"]}
    xFmt="identity"
    yFmt="pct"
    swapXY=true
    title="Student Attendance Rates"
    height=400
/>
```
