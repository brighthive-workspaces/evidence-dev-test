```sql attendance_by_semester
SELECT
    semester,
    avg(attendance_rate) as attendance_rate
FROM
    database_891377033216.academic_performance ap
JOIN
    database_891377033216.students s ON ap.student_id = s.student_id
WHERE
    ap.final_grade = 'A' 
    AND s.major = 'Computer Science'
GROUP BY
    semester
ORDER BY
    semester;
```
