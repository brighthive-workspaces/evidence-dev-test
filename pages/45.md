```sql student_gpa_by_grade
SELECT 
    sp.final_grade, 
    AVG(sp.gpa_points) AS avg_gpa_points 
FROM 
    brighthive_dev_redshift.academic_performance AS sp
GROUP BY 
    sp.final_grade
ORDER BY 
    sp.final_grade
```
