```sql student_attendance
select
    student_id,
    student_name,
    attendance_rate
from
    student_attendance_data
order by
    attendance_rate desc
```

```jsx
<BarChart
    data={student_attendance}
    x=student_name
    y=attendance_rate
    labels=true
    color="#0072B2"
    height=400
    title="Student Attendance Rates"
    swapXY=true
/>
```
