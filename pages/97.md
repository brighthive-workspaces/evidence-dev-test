### SQL Query
```sql student_counts
SELECT 
    s.first_name,
    COUNT(s.first_name) AS name_count
FROM 
    "workspace-database"."brighthive_dev_redshift"."academic_performance" AS a
JOIN 
    "workspace-database"."brighthive_dev_redshift"."students" AS s
ON 
    a.student_id = s.student_id
WHERE 
    a.final_grade = 'A'
GROUP BY 
    s.first_name
ORDER BY 
    name_count DESC;
```
