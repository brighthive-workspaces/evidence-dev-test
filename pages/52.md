```sql
exceptional_students
SELECT 
    s.major,
    COUNT(DISTINCT s.student_id) AS student_count
FROM 
    brighthive_dev_redshift.students s 
JOIN 
    brighthive_dev_redshift.academic_performance a ON s.student_id = a.student_id
WHERE 
    a.final_grade > 85
GROUP BY 
    s.major
ORDER BY 
    student_count DESC;
```
