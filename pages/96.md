```sql major_gpa
SELECT 
    s.major,
    SUM(CAST(ap.gpa_points AS FLOAT)) / COUNT(*) AS average_gpa
FROM 
    brighthive_dev_redshift.students AS s
JOIN 
    brighthive_dev_redshift.academic_performance AS ap
    ON s.student_id = ap.student_id
GROUP BY 
    s.major
ORDER BY 
    average_gpa DESC;
```
