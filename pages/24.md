```sql students_attendance
SELECT
    student_name,
    attendance_rate
FROM
    student_attendance
ORDER BY
    attendance_rate DESC
```

```jsx
<BarChart
    data={students_attendance}
    x=student_name
    y=attendance_rate
    labels=true
    xLabel="Students"
    yLabel="Attendance Rate (%)"
    height=400
    title="Students' Attendance Rates"
    colors={["#006400"]}
/>
```
