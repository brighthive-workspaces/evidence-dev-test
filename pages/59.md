```sql
students_performance
SELECT 
    ap.performance_id, 
    s.first_name 
FROM 
    brighthive_dev_redshift.academic_performance ap 
JOIN 
    brighthive_dev_redshift.students s 
ON 
    ap.student_id = s.student_id;
```
