```sql attendance_rates
select
    student_name,
    count(attended) filter (where attended = true) * 100.0 / count(*) as attendance_rate
from attendance
where class_date between '${inputs.start_date}' and '${inputs.end_date}'
group by student_name
order by attendance_rate desc
```

```jsx
<BarChart
    data={attendance_rates}
    x=student_name
    y=attendance_rate
    labels=true
    type=grouped
    xAxis.title="Student Name"
    yAxis.title="Attendance Rate (%)"
    format={{ y: 'pct2' }}
    colorPalette={['#1f77b4']}
/>
```
