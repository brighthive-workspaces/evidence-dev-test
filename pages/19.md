```sql student_attendance
SELECT
    s.first_name || ' ' || s.last_name AS student_name,
    a.attendance_rate
FROM
    database_891377033216.students s
JOIN
    database_891377033216.academic_performance a
ON
    s.student_id = a.student_id
ORDER BY
    student_name
```
