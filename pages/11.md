```sql student_attendance
SELECT
    student_name,
    attendance_rate
FROM
    student_data
ORDER BY
    attendance_rate DESC;
```

```jsx
<BarChart
    data={student_attendance}
    x=student_name
    y=attendance_rate
    labels=true
    xLabelRotation=45
    title="Student Attendance Rates"
    height=400
    colorPalette={["#4285F4", "#DB4437", "#F4B400", "#0F9D58"]}
/>
```
