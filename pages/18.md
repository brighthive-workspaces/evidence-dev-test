```sql student_attendance
SELECT
    student_name,
    attendance_rate
FROM
    student_attendance_view
ORDER BY
    attendance_rate DESC
```

```jsx
<BarChart
    data={student_attendance}
    x=student_name
    y=attendance_rate
    xLabel="Student Name"
    yLabel="Attendance Rate (%)"
    labels=true
    colorPalette={"#72a0c1"}
    height=350
/>
```
