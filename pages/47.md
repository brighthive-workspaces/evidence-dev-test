```sql gpa_by_student_group
SELECT
    student_group AS student_group,
    SUM(CAST(gpa AS FLOAT)) / COUNT(*) AS average_gpa
FROM
    student_data
GROUP BY
    student_group
ORDER BY
    average_gpa DESC;
```
